## 📦 快速开始示例

```python
# PaddleOCR 快速示例
from paddleocr import PaddleOCR

# 使用轻量模型
ocr = PaddleOCR(
    use_angle_cls=True,
    lang='ch',
    det_model_dir='./models/ch_PP-OCRv4_det_infer',  # 检测
    rec_model_dir='./models/ch_PP-OCRv4_rec_infer',  # 识别
    use_gpu=False  # CPU 运行
)

# 识别视频帧
import cv2
cap = cv2.VideoCapture('video.mp4')

while True:
    ret, frame = cap.read()
    if not ret:
        break
    
    # 裁剪字幕区域（底部 10%）
    h, w = frame.shape[:2]
    subtitle_roi = frame[int(h*0.85):h, :]
    
    # OCR 识别
    result = ocr.ocr(subtitle_roi, cls=True)
    
    if result[0]:
        text = ' '.join([line[1][0] for line in result[0]])
        print(text)
```
